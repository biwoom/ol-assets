## 방법 1. Node.js 내장 서버 + chokidar 감시 (권장)

외부 도구 없이 Node.js만으로 구현합니다. 파일 수정 시 자동 빌드 + 브라우저 자동 새로고침.

### dev-server.js 하나만 추가

```js

// dev-server.js
const fs       = require('fs');
const http     = require('http');
const path     = require('path');
const { exec } = require('child_process');

const PORT = 3000;
let buildPending = false;

// ── 1. 파일 변경 감지 ──
function watchSrc() {
  const dirs = ['src/core', 'src/ui', 'src/styles', 'src'];
  dirs.forEach(dir => {
    if (!fs.existsSync(dir)) return;
    fs.watch(dir, { recursive: true }, (event, filename) => {
      if (!filename || buildPending) return;
      buildPending = true;
      console.log(`\n✎ 변경 감지: ${filename} → 빌드 중...`);
      exec('node build.js', (err, stdout, stderr) => {
        buildPending = false;
        if (err) {
          console.error('✗ 빌드 실패:', stderr);
        } else {
          console.log('✓ 빌드 완료 → 브라우저 새로고침');
          triggerReload();
        }
      });
    });
  });
}

// ── 2. SSE로 브라우저에 새로고침 신호 ──
const clients = new Set();

function triggerReload() {
  clients.forEach(res => res.write('data: reload\n\n'));
}

// ── 3. HTTP 서버 ──
const server = http.createServer((req, res) => {

  // SSE 엔드포인트
  if (req.url === '/__reload') {
    res.writeHead(200, {
      'Content-Type':  'text/event-stream',
      'Cache-Control': 'no-cache',
      'Connection':    'keep-alive',
    });
    res.write('data: connected\n\n');
    clients.add(res);
    req.on('close', () => clients.delete(res));
    return;
  }

  // dist/OL_ATLAS_v0.6.html 서빙 + SSE 스크립트 주입
  const filePath = path.join(__dirname, 'dist', 'OL_ATLAS_v0.6.html');
  if (!fs.existsSync(filePath)) {
    res.writeHead(404);
    res.end('dist 파일 없음. node build.js 먼저 실행하세요.');
    return;
  }

  let html = fs.readFileSync(filePath, 'utf8');

  // 자동 새로고침 스크립트 주입 (</body> 앞에)
  const reloadScript = `
<script>
(function() {
  const es = new EventSource('/__reload');
  es.onmessage = e => { if (e.data === 'reload') location.reload(); };
  es.onerror   = () => setTimeout(() => location.reload(), 1000);
  console.log('[dev] 라이브 리로드 활성화');
})();
</script>`;

  html = html.replace('</body>', reloadScript + '\n</body>');

  res.writeHead(200, { 'Content-Type': 'text/html; charset=utf-8' });
  res.end(html);
});

server.listen(PORT, () => {
  console.log(`\n🚀 개발 서버 시작: http://localhost:${PORT}`);
  console.log('   src/ 파일 수정 시 자동 빌드 + 브라우저 새로고침');
  console.log('   종료: Ctrl+C\n');
  watchSrc();
});
```

### package.json에 scripts 추가

```json
{
  "scripts": {
    "build": "node build.js",
    "dev":   "node dev-server.js"
  }
}
```
